                  Examples¶

                  # Construct Load Balancer
- dimensiondata_load_balancer:
    region: na
    location: NA5
    network_domain: test_network
    name: web_lb01
    port: 80
    protocol: http
    algorith: ROUND_ROBIN
    members:
        - name: webserver1
          port: 8080
          ip: 192.160.0.11
        - name: webserver3
          port: 8080
          ip: 192.160.0.13
    ensure: present
